## 🔌 ধাপ २: ওয়্যারিং সংযোগ

### २.१ ব্রেডবোর্ডে কম্পোনেন্ট সেটআপ

```
ব্রেডবোর্ড লেআউট (শীর্ষ দৃশ্য):

┌────────────────────────────────────┐
│  A  B  C  D  E│F  G  H  I  J      │
│               │                   │
│ +5V────────┬──┼──┐                │
│            │  │  │                │
│ DHT11      │  │  └─ DHT (VCC)     │
│  VCC ──────┤  │                   │
│  GND ──────┼──┴─ GND (ব্লু লাইন) │
│  DATA ─────┼─ GPIO14              │
│            │                      │
│ GND ───────┴─ GND                 │
│                                   │
│ সবুজ এলীডি ─────┬── GPIO4         │
│  দীর্ঘ পা ─────┤                 │
│  Rø (220Ω) ─ GND                 │
│                                   │
│ লাল এলীডি ──────┬── GPIO12        │
│  দীর্ঘ পা ─────┤                 │
│  Rø (220Ω) ─ GND                 │
│                                   │
│ সার্ভো:                          │
│  লাল ───────── +5V                │
│  বাদামি ────── GND                │
│  হলুদ ─────── GPIO13 (PWM)       │
└────────────────────────────────────┘
```

### २.२ বিস্তারিত সংযোগ তালিকা

**DHT11 সেন্সর:**
```
//--------DHT11 পিন 1 (VCC)  → +5V (এসপিHP32 পাওয়ার)------//
DHT11 পিন 2 (DATA) → GPIO14 (স্যাম্পলিং)
DHT11 পিন 4 (GND)  → GND
```

**সবুজ এলীডি (দরজা খোলা সূচক):**
```
লম্বা পা (Anode)   → GPIO4 (মাধ্যমে 220Ω রেজিস্টর)
ছোট পা (Cathode)  → GND
```

**লাল এলীডি (AC স্থিতি):**
```
লম্বা পা (Anode)   → GPIO12 (PWM, মাধ্যমে 220Ω রেজিস্টর)
ছোট পা (Cathode)  → GND

নোট: PWM উজ্জ্বলতা নিয়ন্ত্রণ করবে
  - 32°C এর উপরে: 100% (255)
  - 25-32°C: 30% (~76)
  - 25°C এর নিচে: 0% (বন্ধ)
```

**SG90 সার্ভো মোটর:**
```
লাল (VCC)      → +5V (সরাসরি এসপিHP32,50+ থেকে, বা পাওয়ার ব্যাংক)
বাদামি (GND)   → GND
হলুদ (Signal)  → GPIO13 (PWM, 50Hz)

⚠️ গুরুত্বপূর্ণ: সার্ভো পিক 200mA স্পাইকSark খেতে পারে
ESP32 এর অন-বোর্ড রেগুলেটর এটি পরিচালনা করতে পারে,
কিন্তু USB পাওয়ার অপর্যাপ্ত হলে পাওয়ার ব্যাংক ব্যবহার করুন
```

## 📝 ধাপ ३: ফার্মওয়্যার কোড

### ३.१ মূল স্কেচ (Main Arduino Code)

**স্টেপ এ: লাইব্রেরি ইনক্লুড করুন**
```cpp
#include "DHT.h"
#include <ESP32Servo.h>

// পিন সংজ্ঞা
#define DHTPIN 14          // DHT11 ডেটা পিন
#define GREENLEDPIN 4      // দরজা খোলা এলীডি
#define REDLEDPIN 12       // AC স্থিতি এলীডি (PWM)
#define SERVOPPIN 13       // সার্ভো নিয়ন্ত্রণ পিন
#define DHTTYPE DHT11      // DHT 11

// গ্লোবাল ভেরিয়েবল
DHT dht(DHTPIN, DHTTYPE);
Servo doorServo;
bool roomOccupied = false;
unsigned long doorOpenTime = 0;
const unsigned long DOOR_OPEN_DURATION = 5000; // 5 সেকেন্ড

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n🚀 স্মার্ট রুম অটোমেশন শুরু হচ্ছে...");
  
  // পিন কনফিগারেশন
  pinMode(GREENLEDPIN, OUTPUT);
  pinMode(REDLEDPIN, OUTPUT);
  
  // সার্ভো সেটআপ
  doorServo.attach(SERVOPPIN, 1000, 2000);
  doorServo.write(0);  // বন্ধ অবস্থান
  
  // সেন্সর শুরু করুন
  dht.begin();
  
  Serial.println("✅ সিস্টেম প্রস্তুত");
}

void loop() {
  // বর্তমান তাপমাত্রা এবং আর্দ্রতা পড়ুন
  float humidity = dht.readHumidity();
  float temperature = dht.readTemperature();
  
  // সেন্সর ত্রুটি চেক করুন
  if (isnan(humidity) || isnan(temperature)) {
    Serial.println("⚠️ DHT সেন্সর ত্রুটি!");
    return;
  }
  
  Serial.print("🌡️ তাপমাত্রা: ");
  Serial.print(temperature);
  Serial.print("°C | 💧 আর্দ্রতা: ");
  Serial.print(humidity);
  Serial.println("%");
  
  // দরজার সময়কাল যাচাই করুন
  if (roomOccupied && (millis() - doorOpenTime) > DOOR_OPEN_DURATION) {
    Serial.println("🚪 দরজা বন্ধ হচ্ছে...");
    doorServo.write(0);  // বন্ধ অবস্থায় ফিরিয়ে আনুন
    digitalWrite(GREENLEDPIN, LOW);
    roomOccupied = false;
  }
  
  // রুম দখল সনাক্তকরণ সিমুলেশন
  // (আসল প্রকল্পে এটি ক্যামেরা-ভিত্তিক সনাক্তকরণ দ্বারা প্রতিস্থাপিত হবে)
  if (checkFaceDetection()) {  // কাস্টম ফাংশন, নীচে দেখুন
    handleRoomEntry();
  }
  
  // AC লজিক (শুধুমাত্র যদি কক্ষ দখল করা হয়)
  if (roomOccupied) {
    controlACIndicator(temperature);
  } else {
    // রুম খালি → লাল এলীডি বন্ধ
    analogWrite(REDLEDPIN, 0);
  }
  
  delay(2000);  // প্রতি 2 সেকেন্ডে পড়ুন
}

// AC সূচক নিয়ন্ত্রণ করুন
void controlACIndicator(float temp) {
  int brightness;
  
  if (temp > 32.0) {
    brightness = 255;  // 100%
    Serial.println("🔴 AC পূর্ণ শক্তিতে (গরম)");
  } else if (temp >= 25.0 && temp <= 32.0) {
    brightness = 76;   // ~30%
    Serial.println("🟠 AC অর্ধ শক্তিতে (মধ্যম)");
  } else {
    brightness = 0;    // বন্ধ
    Serial.println("🟢 AC বন্ধ (শীতল)");
  }
  
  analogWrite(REDLEDPIN, brightness);
}

// মুখ সনাক্তকরণ (সিমুলেশন)
bool checkFaceDetection() {
  // দ্রুত প্রোটোটাইপিং এর জন্য:
  // সিরিয়াল ইনপুটে 'f' পাঠান দরজা খোলার জন্য
  if (Serial.available()) {
    char cmd = Serial.read();
    if (cmd == 'f') {
      return true;
    }
  }
  return false;
}

// রুম প্রবেশ হ্যান্ডলার
void handleRoomEntry() {
  Serial.println("✅ মুখ সনাক্ত হয়েছে!");
  Serial.println("🟢 সবুজ এলীডি চালু");
  Serial.println("🚪 দরজা খুলছে...");
  
  // সবুজ এলীডি চালু করুন
  digitalWrite(GREENLEDPIN, HIGH);
  
  // দরজা খুলুন
  doorServo.write(90);  // খোলার কোণ
  
  // অবস্থা সেট করুন
  roomOccupied = true;
  doorOpenTime = millis();
}
```

```cpp
#include <WiFi.h>

const char* ssid = "আপনার_WiFi";
const char* password = "পাসওয়ার্ড";

void setupWiFi() {
  WiFi.begin(ssid, password);
  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 20) {
    delay(500);
    Serial.print(".");
    attempts++;
  }
  Serial.println(WiFi.status() == WL_CONNECTED ? "\n✅ WiFi সংযুক্ত" : "\n❌ WiFi ব্যর্থ");
}
```
